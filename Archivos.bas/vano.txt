Attribute VB_Name = "vano"
Function vano(ByRef rayon As Double) As Double
Dim m As Integer
m = 3
If rayon < 0 Then
    rayon = rayon * (-1)
End If
' buscar en que fila de la hoja 2 contiene el radio actual y elegir el vano correspondiente
If Not rayon = 0 Then
    While rayon < Sheets(2).Cells(m, 3).Value _
    Or rayon > Sheets(2).Cells(m, 2).Value
    m = m + 1
    Wend
End If
    vano = Sheets(2).Cells(m, 1).Value
End Function
